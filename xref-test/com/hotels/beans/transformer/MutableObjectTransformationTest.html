121     /**
122      * Test that a given field transformer function is applied only to a specific field even if there are other ones with same name.
123      */
124     @Test
125     public void testFieldTransformationIsAppliedOnlyToASpecificField() {
126         //GIVEN
127         String namePrefix = "prefix-";
128         FieldTransformer<String, String> nameTransformer = new FieldTransformer<>("nestedObject.name", val -> namePrefix + val);
129 
130         //WHEN
131         MutableToFoo actual = underTest
132                 .withFieldTransformer(nameTransformer)
133                 .transform(fromFoo, MutableToFoo.class);
134 
135         //THEN
136         assertEquals(fromFoo.getName(), actual.getName());
137         assertEquals(namePrefix + fromFoo.getNestedObject().getName(), actual.getNestedObject().getName());
138         underTest.resetFieldsTransformer();
139     }

141     /**
142      * Test that a given field transformer function is applied to all field matching the given name when {@code flatFieldNameTransformation} is set to true.
143      */
144     @Test
145     public void testFieldTransformationIsAppliedToAllMatchingFields() {
146         //GIVEN
147         String namePrefix = "prefix-";
148         FieldTransformer<String, String> nameTransformer = new FieldTransformer<>(NAME_FIELD_NAME, val -> namePrefix + val);
149 
150         //WHEN
151         MutableToFoo actual = underTest
152                 .setFlatFieldNameTransformation(true)
153                 .withFieldTransformer(nameTransformer)
154                 .transform(fromFoo, MutableToFoo.class);
155 
156         //THEN
157         assertEquals(namePrefix + fromFoo.getName(), actual.getName());
158         assertEquals(namePrefix + fromFoo.getNestedObject().getName(), actual.getNestedObject().getName());
159         underTest.resetFieldsTransformer();
160     }

178     /**
179      * Test that a bean containing a field not existing in the source object, but with a transformer function defined for such object is correctly copied.

181     @Test
182     public void testThatAnyTypeOfBeanContainsANotExistingFieldInTheSourceObjectIsCorrectlyCopiedThroughTransformerFunctions() {
183         //GIVEN
184         FromFooSimple fromFooSimple = new FromFooSimple(NAME, ID);
185         FieldTransformer<Object, Integer> ageFieldTransformer = new FieldTransformer<>(AGE_FIELD_NAME, val -> AGE);
186 
187         //WHEN

189         MutableToFooNotExistingFields mutableObjectBean = underTest.transform(fromFooSimple, MutableToFooNotExistingFields.class);

195     /**
196      * Test transformation field with field transformer.
197      * @param testCaseDescription the test case description
198      * @param fieldToTransform the name of the field on which apply the transformation
199      * @param transformationResult the value to return after the transformation
200      */
201     @Test(dataProvider = "dataTransformationTesting")
202     public void testTransformationWithFieldTransformationWorksProperly(final String testCaseDescription, final String fieldToTransform, final Object transformationResult) {
203         //GIVEN
204         FromFooSimple fromFooSimple = new FromFooSimple(NAME, ID);
205         FieldTransformer<Object, Object> fieldTransformer = new FieldTransformer<>(fieldToTransform, val -> transformationResult);
206 
207         //WHEN
208         underTest.withFieldTransformer(fieldTransformer);
209         MutableToFooSimple actual = underTest.transform(fromFooSimple, MutableToFooSimple.class);
210 
211         //THEN
212         assertThat(actual, hasProperty(fieldToTransform, equalTo(transformationResult)));
213         underTest.removeFieldTransformer(fieldToTransform);
214     }
