85      /**
86       * Invokes the method.
87       * @param method the method to be invoked.
88       * @param target the class on which invoke the method
89       * @param args the method parameters
90       * @return the method result
91       */
92      protected Object invokeMethod(final Method method, final Object target, final Object... args) {
93          try {
94              return method.invoke(target, args);
95          } catch (final Exception e) {
96              handleReflectionException(e);
97              throw new IllegalStateException(e);
98          }
99      }

101     /**
102      * Checks if the given method is a setter.
103      * @param method the method to be checked
104      * @return true if the method is a setter method, false otherwise
105      */
106     public boolean isSetter(final Method method) {
107         final String cacheKey = "IsSetter-" + method.getDeclaringClass().getName() + '-' + method.getName();
108         return CACHE_MANAGER.getFromCache(cacheKey, Boolean.class).orElseGet(() -> {
109             boolean res = isPublic(method.getModifiers())
110                     && method.getName().matches(SETTER_METHOD_NAME_REGEX)
111                     && method.getParameterTypes().length == 1
112                     && method.getReturnType().equals(void.class);
113             CACHE_MANAGER.cacheObject(cacheKey, res);
114             return res;
115         });
116     }

128     /**
129      * Gets the value of a field through getter method.
130      * @param target the field's class
131      * @param fieldName the field name
132      * @param fieldType the field type
133      * @return the field value
134      */
135     @SuppressWarnings("unchecked")
136     public Object getFieldValue(final Object target, final String fieldName, final Class<?> fieldType) {
137         Object fieldValue = getRealTarget(target);
138         for (String currFieldName : fieldName.split(DOT_SPLIT_REGEX)) {
139             if (fieldValue == null) {
140                 break;
141             }
142             try {
143                 fieldValue = getGetterMethodFunction(fieldValue.getClass(), currFieldName).apply(fieldValue);
144             } catch (final ClassCastException | MissingMethodException | InvalidBeanException e) {
145                 fieldValue = getFieldValueDirectAccess(fieldValue, currFieldName);
146             } catch (final MissingFieldException e) {
147                 fieldValue = invokeMethod(getGetterMethod(fieldValue.getClass(), currFieldName, fieldType), fieldValue);
148             }
149         }
150         return fieldValue;
151     }

153     /**
154      * Returns the getter method for the given field.
155      * @param fieldClass the field's class
156      * @param fieldName the field name
157      * @param fieldType the field type
158      * @return the getter method
159      */
160     private Method getGetterMethod(final Class<?> fieldClass, final String fieldName, final Class<?> fieldType) {
161         final String cacheKey = "GetterMethod-" + fieldClass.getName() + '-' + fieldName;
162         return CACHE_MANAGER.getFromCache(cacheKey, Method.class).orElseGet(() -> {
163             try {
164                 Method method = fieldClass.getMethod(getGetterMethodPrefix(fieldType) + capitalize(fieldName));
165                 method.setAccessible(true);
166                 CACHE_MANAGER.cacheObject(cacheKey, method);
167                 return method;
168             } catch (NoSuchMethodException e) {
169                 throw new MissingFieldException(fieldClass.getName() + " hasn't a field called: " + fieldName + ".");
170             }
171         });
172     }

174     /**
175      * Returns the getter method for the given field.
176      * @param fieldClass the field's class
177      * @param fieldName the field name
178      * @return the getter method
179      */
180     private Function getGetterMethodFunction(final Class<?> fieldClass, final String fieldName) {
181         final String cacheKey = "getGetterMethodFunction-" + fieldClass.getName() + '-' + fieldName;
182         return CACHE_MANAGER.getFromCache(cacheKey, Function.class).orElseGet(() -> {
183             Function function;
184             try {
185                 Class<?> fieldType = getDeclaredFieldType(fieldName, fieldClass);
186                 MethodHandles.Lookup privateLookupIn = privateLookupIn(fieldClass, METHOD_HANDLES_LOOKUP);
187                 CallSite site = metafactory(privateLookupIn,
188                         "apply",
189                         methodType(Function.class),
190                         methodType(Object.class, Object.class),
191                         privateLookupIn.findVirtual(fieldClass, getGetterMethodPrefix(fieldType) + capitalize(fieldName), methodType(fieldType)),
192                         methodType(fieldType, fieldClass));
193                 function = (Function) site.getTarget().invokeExact();
194             } catch (NoSuchFieldException | MissingFieldException e) {
195                 throw new MissingFieldException(e.getMessage());
196             } catch (NoSuchMethodException e) {
197                 throw new MissingMethodException();
198             } catch (Throwable e) {
199                 throw new InvalidBeanException(e);
200             }
201             CACHE_MANAGER.cacheObject(cacheKey, function);
202             return function;
203         });
204     }

251     private Object getFieldValueDirectAccess(final Object target, final String fieldName) {
252         try {
253             Field field = getDeclaredField(fieldName, target.getClass());
254             return field.get(target);

259             throw new IllegalStateException(e);

263     /**
264      * Return the field of the given class.
265      * @param fieldName the name of the field to retrieve.
266      * @param targetClass the field's class
267      * @return the field corresponding to the given name.
268      */
269     public Field getDeclaredField(final String fieldName, final Class<?> targetClass) {
270         final String cacheKey = "ClassDeclaredField-" + targetClass.getName() + "-" + fieldName;
271         return CACHE_MANAGER.getFromCache(cacheKey, Field.class).orElseGet(() -> {
272             Field field;
273             try {
274                 field = targetClass.getDeclaredField(fieldName);
275                 field.setAccessible(true);
276             } catch (NoSuchFieldException e) {
277                 Class<?> superclass = targetClass.getSuperclass();
278                 if (!superclass.equals(Object.class)) {
279                     field = getDeclaredField(fieldName, superclass);
280                 } else {
281                     throw new MissingFieldException(targetClass.getName() + " does not contain field: " + fieldName);
282                 }
283             } catch (final Exception e) {
284                 handleReflectionException(e);
285                 throw new IllegalStateException(e);
286             }
287             CACHE_MANAGER.cacheObject(cacheKey, field);
288             return field;
289         });
290     }

293      * Return the class of the given field.
294      * @param fieldName the name of the filed to retrieve.
295      * @param clazz the field's class
296      * @return the class field of the given field.
297      */
298     public Class<?> getDeclaredFieldType(final String fieldName, final Class<?> clazz) {
299         final String cacheKey = "FieldType-" + clazz.getName() + "-" + fieldName;
300         return CACHE_MANAGER.getFromCache(cacheKey, Class.class).orElseGet(() -> {
301             Class<?> fieldType;
302             Field field;
303             try {
304                 field = clazz.getDeclaredField(fieldName);
305             } catch (NoSuchFieldException e) {
306                 Class<?> superclass = clazz.getSuperclass();
307                 if (!superclass.equals(Object.class)) {
308                     field = getDeclaredField(fieldName, superclass);
309                 } else {
310                     throw new MissingFieldException(clazz.getName() + " does not contain field: " + fieldName);
311                 }
312             } catch (final Exception e) {
313                 handleReflectionException(e);
314                 throw new IllegalStateException(e);
315             }
316             fieldType = field.getType();
317             CACHE_MANAGER.cacheObject(cacheKey, fieldType);
318             return fieldType;
319         });
320     }

323      * Set the value of a field.
324      * @param target the field's class
325      * @param field the field to set
326      * @param fieldValue the value to set
327      */
328     public void setFieldValue(final Object target, final Field field, final Object fieldValue) {
329         try {
330             setFieldValueWithoutSetterMethod(target, field, fieldValue);
331         } catch (final IllegalArgumentException e) {
332             throw e;
333         } catch (final Exception e) {
334             invokeMethod(getSetterMethodForField(target.getClass(), field.getName(), field.getType()), target, fieldValue);
335         }
336     }

353     /**
354      * If an object is encapsulated into a type: {@link Optional} the contained object is returned.
355      * @param target the object to check
356      * @return the encapsulated object (if any)
357      */
358     private Object getRealTarget(final Object target) {
359         AtomicReference<Object> realTarget = new AtomicReference<>(target);
360         if (isOptionalType(target)) {
361             ((Optional<?>) target).ifPresent(realTarget::set);
362         }
363         return realTarget.get();
364     }

382         return CACHE_MANAGER.getFromCache(cacheKey, String.class).orElseGet(() -> {

456      * Retrieves the generic class of the objects contained in the map.
457      * It could be {@link ItemType} in case the object is primitive or is a collection
458      * It could be {@link MapType} in case the object is a {@link Map}
459      * @param fieldType the field containing the generic
460      * @param declaringClass the class containing the field
461      * @param fieldName the field name
462      * @return the generic class of the objects contained in the map.

464     private MapElemType getMapElemType(final Type fieldType, final String declaringClass, final String fieldName) {
465         final String cacheKey = "MapElemType-" + declaringClass + "-" + fieldType.getTypeName() + '-' + fieldName;
466         return CACHE_MANAGER.getFromCache(cacheKey, MapElemType.class).orElseGet(() -> {
467             final Class<?> argumentTypeClass = getArgumentTypeClass(fieldType, declaringClass, fieldName, false);
468             final MapElemType res;
469             if (Map.class.isAssignableFrom(argumentTypeClass)) {
470                 res = getMapGenericType(fieldType, declaringClass, fieldName);
471             } else {
472                 res = buildItemType(fieldType, declaringClass, fieldName);
473             }
474             CACHE_MANAGER.cacheObject(cacheKey, res);
475             return res;

493     /**
494      * Gets the class of a given object.
495      * @param argument the object from which the class has to be retrieved
496      * @param declaringClass the class containing the field of which the argument belongs to
497      * @param fieldName the field name of which the argument belongs to
498      * @param getNestedGenericClass if true it retrieves the class of the object generic (if any). i.e. {@code argument = List<String>;}
499      *                              returns {@code String}, if false returns {@code List}
500      * @return the given argument class type
501      */
502     public Class<?> getArgumentTypeClass(final Object argument, final String declaringClass, final String fieldName, final boolean getNestedGenericClass) {
503         final boolean argumentIsTypeClass = argument instanceof Class;
504         final String cacheKey = "ArgumentTypeClass-" + declaringClass + "-" + fieldName + "-" + getNestedGenericClass + "-"
505                 + "-" + (argumentIsTypeClass || !List.class.isAssignableFrom(argument.getClass()) ? argument : argument.getClass());
506         return CACHE_MANAGER.getFromCache(cacheKey, Class.class)
507                 .map(atc -> atc == EmptyValue.class ? null : atc)
508                 .orElseGet(() -> {
509                     Class<?> res = null;
510                     if (argumentIsTypeClass) {
511                         res = getNestedGenericClass ? null : (Class<?>) argument;
512                     } else if (ParameterizedType.class.isAssignableFrom(argument.getClass())) {
513                         res = getNestedGenericClass
514                                 ? getArgumentTypeClass(((ParameterizedType) argument).getActualTypeArguments()[0], declaringClass, fieldName, false)
515                                 : (Class<?>) ((ParameterizedType) argument).getRawType();
516                     }
517                     CACHE_MANAGER.cacheObject(cacheKey, res, EmptyValue.class);
518                     return res;
519                 });
520     }

536     /**
537      * Throws the right exception.
538      * @param ex the exception
539      */
540     void handleReflectionException(final Exception ex) {
541         if (ex instanceof NoSuchMethodException) {
542             throw new MissingMethodException("Method not found: " + ex.getMessage());
543         } else if (ex instanceof IllegalAccessException) {
544             throw new IllegalStateException("Could not access method: " + ex.getMessage(), ex);
545         } else {
546             if (ex instanceof RuntimeException) {
547                 throw (RuntimeException) ex;
548             } else {
549                 throw new UndeclaredThrowableException(ex);
550             }
551         }
